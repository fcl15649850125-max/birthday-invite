    <script>
        var myMusic = document.getElementById("bgMusic");
        var btn = document.getElementById("music-btn");
        var isPlaying = false;

        function toggleMusic() {
            if (isPlaying) {
                myMusic.pause();
                btn.innerHTML = "🎵 音乐已暂停 (点击播放)";
            } else {
                myMusic.play();
                btn.innerHTML = "🎶 音乐播放中 (点击暂停)";
            }
            isPlaying = !isPlaying;
        }
    </script>
